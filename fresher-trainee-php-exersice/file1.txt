Make Script to Withdraw Amount From ATM Machine

1) Set the amount in a variable , for example 10,000
2) Withdraw Amount in 50,100 and 500 Rupees note
3) When You withdraw you need show noted withdraw in particular amount , form if I withdraw 5000 in 50 rupees note than output will 50 x 100
4) Also show remaining amount left in ATM



<script>
 
// Javascript program to accept an amount
// and count number of notes
 
    // function to count and
    // print currency notes
    function countCurrency(amount)
    {
        let notes = [ 2000, 500, 200, 100, 50, 20, 10, 5, 1 ];
        let noteCounter = Array(9).fill(0);
        
        // count notes using Greedy approach
        for (let i = 0; i < 9; i++) {
            if (amount >= notes[i]) {
                noteCounter[i] = Math.floor(amount / notes[i]);
                amount = amount - noteCounter[i] * notes[i];
            }
        }
        
        // Print notes
       document.write("Currency Count ->" + "<br/>");
        for (let i = 0; i < 9; i++) {
            if (noteCounter[i] != 0) {
                document.write(notes[i] + " : "
                    + noteCounter[i] + "<br/>");
            }
        }
    }
     
 
// driver code
 
    let amount = 868;
    countCurrency(amount);
     
</script>
